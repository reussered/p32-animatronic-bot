void setPowerMode(PowerState mode) {
    switch(mode) {
        case SLEEP_MODE:
            disableNonEssentialMotors();
            dimAllDisplays(10);
            setStepperCurrent(0.5); // Reduce holding current
            break;
        case ACTIVE_MODE:
            enableAllSystems();
            setStepperCurrent(1.0);
            break;
    }
}
